\section*{POST /instance}

\textbf{Request body}

\begin{tabularx}{\textwidth}{>{\raggedright\arraybackslash}p{3.2cm} >{\raggedright\arraybackslash}p{1.6cm} X}
\toprule
Field & Type & Rules \\
\midrule
\texttt{project\_id} & string & required, non-empty \\
\texttt{image} & string & required, non-empty \\
\texttt{cpu\_millicores} & integer & required, must be $>$ 0 \\
\texttt{memory\_mb} & integer & required, must be $>$ 0 \\
\texttt{name} & string & optional, auto-generated if missing \\
\bottomrule
\end{tabularx}

\textbf{Success (201)}

Returns a full instance object:
\texttt{id}, \texttt{project\_id}, \texttt{docker\_container\_id}, \texttt{name}, \texttt{image},
\texttt{status}, \texttt{cpu\_millicores}, \texttt{memory\_mb}, \texttt{network\_name},
\texttt{last\_error}, \texttt{created\_at}, \texttt{updated\_at}.

\textbf{Errors}
\texttt{400} invalid fields or Docker create failure; \texttt{409} quota denied.

\textbf{Example request}
\begin{verbatim}
POST /instance
Content-Type: application/json

{
  "project_id": "project-a",
  "image": "nginx:alpine",
  "cpu_millicores": 250,
  "memory_mb": 128,
  "name": "web-a"
}
\end{verbatim}

\textbf{Example response (201)}
\begin{verbatim}
{
  "id": "8f75d4a2-6bce-4e8a-9f0f-7dc16d5f5c6e",
  "project_id": "project-a",
  "docker_container_id": "3ca6f4e91...",
  "name": "web-a",
  "image": "nginx:alpine",
  "status": "running",
  "cpu_millicores": 250,
  "memory_mb": 128,
  "network_name": "campuscloud-project-project-a",
  "last_error": null,
  "created_at": "2026-03-16T08:00:00+00:00",
  "updated_at": "2026-03-16T08:00:01+00:00"
}
\end{verbatim}

\section*{POST /instance/\{instance\_id\}/stop}

\textbf{Path parameter}
\texttt{instance\_id} (string, required).

\textbf{Success (200)}
Returns the updated instance object with \texttt{status} set to \texttt{stopped} when a running
container is stopped. If status is already \texttt{stopped}, \texttt{failed}, or \texttt{deleted}, the
endpoint returns the current instance object.

\textbf{Errors}
\texttt{404} instance not found; \texttt{502} Docker stop failure.

\textbf{Example request}
\begin{verbatim}
POST /instance/8f75d4a2-6bce-4e8a-9f0f-7dc16d5f5c6e/stop
\end{verbatim}

\textbf{Example response (200)}
\begin{verbatim}
{
  "id": "8f75d4a2-6bce-4e8a-9f0f-7dc16d5f5c6e",
  "project_id": "project-a",
  "status": "stopped",
  "docker_container_id": "3ca6f4e91..."
}
\end{verbatim}

\section*{DELETE /instance/\{instance\_id\}}

\textbf{Path parameter}
\texttt{instance\_id} (string, required).

\textbf{Success (204)}
No response body. Deletion is idempotent: missing instance or already deleted instance still returns
\texttt{204}.

\textbf{Errors}
\texttt{502} Docker remove failure.

\textbf{Example request}
\begin{verbatim}
DELETE /instance/8f75d4a2-6bce-4e8a-9f0f-7dc16d5f5c6e
\end{verbatim}

\section*{GET /instances}

\textbf{Query parameter}
\texttt{project\_id} (string, required, non-empty).

\textbf{Success (200)}
Returns an array of instance objects for that project. Rows with status \texttt{deleted} are excluded.

\textbf{Errors}
\texttt{400} if \texttt{project\_id} is missing.

\textbf{Example request}
\begin{verbatim}
GET /instances?project_id=project-a
\end{verbatim}

\textbf{Example response (200)}
\begin{verbatim}
[
  {
    "id": "8f75d4a2-6bce-4e8a-9f0f-7dc16d5f5c6e",
    "project_id": "project-a",
    "status": "running",
    "name": "web-a",
    "network_name": "campuscloud-project-project-a"
  }
]
\end{verbatim}
